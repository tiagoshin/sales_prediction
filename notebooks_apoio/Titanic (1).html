def cleanout05_fare(x):
    if x < df["Fare"].quantile(q=0.05):
        x = np.nan
    else:
        if x > df["Fare"].quantile(q=0.95):
            x = np.nan
        else: 
            x = x
    return x

def nan_fare_bc(dados):
    fare = dados[0]
    pclass = dados[1]
    
    if pd.isnull(fare) == True:
        if pclass == 1:
            return df["Fare"][df['Pclass'] == 1].mean()
        else:
            if pclass == 2:
                return df["Fare"][df['Pclass'] == 2].mean()
            else:
                return df["Fare"][df['Pclass'] == 3].mean()
    else:
        return fare
